26// utility method to load data into a wide type
27template<eve::simd_value T, std::random_access_iterator I, typename F>
28 requires std::is_invocable_v<F, std::iter_value_t<I>>
29auto inline load(I iter, F&& func)
30{
31 return [&]<std::size_t... Idx>(std::index_sequence<Idx...>) -> auto
32 { return T {std::forward<F>(func)(*(iter + Idx))...}; }(std::make_index_sequence<T::size()> {});
33}
34
35// binary projection overload: applies func(a, b) element-wise across two iterators
36template<eve::simd_value T, std::random_access_iterator I, std::random_access_iterator J, typename F>
37 requires std::is_invocable_v<F, std::iter_value_t<I>, std::iter_value_t<J>>
38auto inline load(I iter1, J iter2, F&& func)
39{
40 return [&]<std::size_t... Idx>(std::index_sequence<Idx...>) -> auto
41 { return T {std::forward<F>(func)(*(iter1 + Idx), *(iter2 + Idx))...}; }(std::make_index_sequence<T::size()> {});
42}
43
44// utility method to advance a set of iterators
45template<typename Distance, typename... Iters>
46auto inline advance(Distance d, Iters&... iters) -> void
47{
48 (std::advance(iters, d), ...);
49}

83template<std::floating_point T, stats Stats = stats::variance, std::random_access_iterator I, typename F = std::identity>
84 requires concepts::arithmetic_projection<F, std::iter_value_t<I>>
85inline auto accumulate(I first, I last, F&& f = F {}) noexcept -> univariate_statistics
86{
87 using wide = eve::wide<T>;
88 auto constexpr s {wide::size()};
89 auto const n {std::distance(first, last)};
90 auto const m = n - (n % s);
91
92 if (n < s) {
93 univariate_accumulator<T, Stats> scalar_acc;
94 for (; first < last; ++first) {
95 scalar_acc(std::invoke(std::forward<F>(f), *first));
96 }
97 return univariate_statistics(scalar_acc);
98 }
99
100 univariate_accumulator<wide, Stats> acc;
101 for (size_t i = 0; i < m; i += s) {
102 acc(detail::load<wide>(first, std::forward<F>(f)));
103 detail::advance(s, first);
104 }
105
106 // gather the remaining values with a scalar accumulator
107 if (m < n) {
108 auto [sw, sx, sxx] = acc.stats();
109 auto scalar_acc = univariate_accumulator<T, Stats>::load_state(sw, sx, sxx);
110 for (; first < last; ++first) {
111 scalar_acc(std::invoke(std::forward<F>(f), *first));
112 }
113 return univariate_statistics(scalar_acc);
114 }
115 return univariate_statistics(acc);
116}

487template<std::floating_point T, std::contiguous_iterator I, std::contiguous_iterator J>
488inline auto r2_score(I first1, I last1, J first2) noexcept -> double
489{
490 using wide = eve::wide<T>;
491 auto constexpr s {wide::size()};
492 auto const n {std::distance(first1, last1)};
493 auto const m {n - (n % s)};
494
495 univariate_accumulator<wide, stats::sum> wx;
496 univariate_accumulator<wide, stats::variance> wy;
497 for (auto i = 0; i < m; i += s) {
498 wide y_true {std::to_address(first1)};
499 wide y_pred {std::to_address(first2)};
500 wx(eve::sqr(y_true - y_pred));
501 wy(y_true);
502 detail::advance(s, first1, first2);
503 }
504
505 // use scalar accumulators for the remaining values
506 auto sx = univariate_accumulator<T, stats::sum>::load_state(wx.stats());
507 auto sy = univariate_accumulator<T, stats::variance>::load_state(wy.stats());
508
509 for (; first1 < last1; ++first1, ++first2) {
510 sx(eve::sqr(*first1 - *first2));
511 sy(*first1);
512 }
513
514 auto const rss = univariate_statistics(sx).sum;
515 auto const tss = univariate_statistics(sy).ssr;
516
517 return tss < std::numeric_limits<double>::epsilon() ? std::numeric_limits<double>::lowest() : 1.0 - (rss / tss);
518}

786template<std::floating_point T, std::contiguous_iterator I, std::contiguous_iterator J>
787inline auto mean_absolute_percentage_error(I first1, I last1, J first2) noexcept -> double
788{
789 using wide = eve::wide<T>;
790 auto constexpr s {wide::size()};
791 auto const n {std::distance(first1, last1)};
792 auto const m {n - n % s};
793
794 auto constexpr eps {std::numeric_limits<T>::epsilon()};
795
796 univariate_accumulator<wide, stats::mean> we;
797 for (auto i = 0; i < m; i += s) {
798 wide y_true {std::to_address(first1)};
799 wide y_pred {std::to_address(first2)};
800 we(eve::abs(y_true - y_pred) / eve::max(eps, eve::abs(y_true)));
801 detail::advance(s, first1, first2);
802 }

937template<std::floating_point T, std::contiguous_iterator I, std::contiguous_iterator J>
938inline auto gaussian_neg_likelihood_loss(I first1, I last1, J first2, T sigma) noexcept
939 -> double
940{
941 using wide = eve::wide<T>;
942 auto constexpr s {wide::size()};
943 auto const n {std::distance(first1, last1)};
944 auto const m {n - n % s};
945
946 univariate_accumulator<wide, stats::sum> we;
947 for (auto i = 0; i < m; i += s) {
948 wide y_true {std::to_address(first1)};
949 wide y_pred {std::to_address(first2)};
950 we(eve::sqr(y_true - y_pred));
951 detail::advance(s, first1, first2);
952 }
953
954 // use scalar accumulators for the remaining values
955 auto se = univariate_accumulator<T, stats::sum>::load_state(we.stats());
956 for (; first1 < last1; ++first1, ++first2) {
957 se(eve::sqr(*first1 - *first2));
958 }
959 auto const ssr = univariate_statistics(se).sum;
960 auto const pi = std::numbers::pi_v<double>;
961 return 0.5 * static_cast<double>(n) * std::log(2.0 * pi)
962 + static_cast<double>(n) * std::log(static_cast<double>(sigma))
963 + ssr / (2.0 * static_cast<double>(sigma) * static_cast<double>(sigma));
964}
