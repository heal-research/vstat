24// utility method to load data into a wide type
25template<eve::simd_value T, std::input_iterator I, typename F>
26 requires std::is_invocable_v<F, std::iter_value_t<I>>
27auto inline load(I iter, F&& func)
28{
29 return [&]<std::size_t... Idx>(std::index_sequence<Idx...>) -> auto
30 { return T {std::forward<F>(func)(*(iter + Idx))...}; }(std::make_index_sequence<T::size()> {});
31}
32
33// utility method to advance a set of iterators
34template<typename Distance, typename... Iters>
35auto inline advance(Distance d, Iters&... iters) -> void
36{
37 (std::advance(iters, d), ...);
38}

72template<std::floating_point T, std::input_iterator I, typename F = std::identity>
73 requires concepts::arithmetic_projection<F, std::iter_value_t<I>>
74inline auto accumulate(I first, std::sized_sentinel_for<I> auto last, F&& f = F {}) noexcept -> univariate_statistics
75{
76 using wide = eve::wide<T>;
77 auto constexpr s {wide::size()};
78 auto const n {std::distance(first, last)};
79 auto const m = n - (n % s);
80
81 if (n < s) {
82 univariate_accumulator<T> scalar_acc;
83 for (; first < last; ++first) {
84 scalar_acc(std::invoke(std::forward<F>(f), *first));
85 }
86 return univariate_statistics(scalar_acc);
87 }
88
89 univariate_accumulator<wide> acc;
90 for (size_t i = 0; i < m; i += s) {
91 acc(detail::load<wide>(first, std::forward<F>(f)));
92 detail::advance(s, first);
93 }
94
95 // gather the remaining values with a scalar accumulator
96 if (m < n) {
97 auto [sw, sx, sxx] = acc.stats();
98 auto scalar_acc = univariate_accumulator<T>::load_state(sw, sx, sxx);
99 for (; first < last; ++first) {
100 scalar_acc(std::invoke(std::forward<F>(f), *first));
101 }
102 return univariate_statistics(scalar_acc);
103 }
104 return univariate_statistics(acc);
105}

176template<std::floating_point T,
177 std::input_iterator I,
178 std::input_iterator J,
179 typename BinaryOp,
180 typename F1 = std::identity,
181 typename F2 = std::identity>
182 requires std::is_invocable_v<F1, std::iter_value_t<I>> and std::is_invocable_v<F2, std::iter_value_t<J>>
183 and std::is_invocable_v<BinaryOp,
184 std::invoke_result_t<F1, std::iter_value_t<I>>,
185 std::invoke_result_t<F2, std::iter_value_t<J>>>
186 and concepts::arithmetic_projection<BinaryOp,
187 std::invoke_result_t<F1, std::iter_value_t<I>>,
188 std::invoke_result_t<F2, std::iter_value_t<J>>>
189inline auto accumulate(I first1,
190 std::sized_sentinel_for<I> auto last1,
191 J first2,
192 BinaryOp&& op = BinaryOp {},
193 F1&& f1 = F1 {},
194 F2&& f2 = F2 {}) noexcept -> univariate_statistics
195{
196 using wide = eve::wide<T>;
197 auto constexpr s {wide::size()};
198 auto const n {std::distance(first1, last1)};
199 auto const m = n - n % s;
200
201 auto f = [&](auto a, auto b)
202 {
203 return std::invoke(
204 std::forward<BinaryOp>(op), std::invoke(std::forward<F1>(f1), a), std::invoke(std::forward<F2>(f2), b));
205 };
206
207 if (n < s) {
208 univariate_accumulator<T> scalar_acc;
209 for (; first1 < last1; ++first1, ++first2) {
210 scalar_acc(f(*first1, *first2));
211 }
212 return univariate_statistics(scalar_acc);
213 }
214
215 std::array<T, s> x;
216 univariate_accumulator<wide> acc;
217 for (size_t i = 0; i < m; i += s) {
218 std::transform(first1, first1 + s, first2, x.begin(), f);
219 acc(wide {x.data()});
220 detail::advance(s, first1, first2);
221 }
222
223 // use a scalar accumulator to gather the remaining values
224 if (m < n) {
225 auto [sw, sx, sxx] = acc.stats();
226 auto scalar_acc = univariate_accumulator<T>::load_state(sw, sx, sxx);
227 for (; first1 < last1; ++first1, ++first2) {
228 scalar_acc(f(*first1, *first2));
229 }
230 return univariate_statistics(scalar_acc);
231 }
232 return univariate_statistics(acc);
233}

474template<std::floating_point T, std::input_iterator I, std::input_iterator J>
475inline auto r2_score(I first1, std::sentinel_for<I> auto last1, J first2) noexcept -> double
476{
477 using wide = eve::wide<T>;
478 auto constexpr s {wide::size()};
479 auto const n {std::distance(first1, last1)};
480 auto const m {n - (n % s)};
481
482 univariate_accumulator<wide> wx;
483 univariate_accumulator<wide> wy;
484 for (auto i = 0; i < m; i += s) {
485 wide y_true {std::to_address(first1)};
486 wide y_pred {std::to_address(first2)};
487 wx(eve::sqr(y_true - y_pred));
488 wy(y_true);
489 detail::advance(s, first1, first2);
490 }
491
492 // use scalar accumulators for the remaining values
493 auto sx = univariate_accumulator<T>::load_state(wx.stats());
494 auto sy = univariate_accumulator<T>::load_state(wy.stats());
495
496 for (; first1 < last1; ++first1, ++first2) {
497 sx(eve::sqr(*first1 - *first2));
498 sy(*first1);
499 }
500
501 auto const rss = univariate_statistics(sx).sum;
502 auto const tss = univariate_statistics(sy).ssr;
503
504 return tss < std::numeric_limits<double>::epsilon() ? std::numeric_limits<double>::lowest() : 1.0 - (rss / tss);
505}

820 univariate_accumulator<wide> we;
821 for (auto i = 0; i < m; i += s) {
822 wide y_true {std::to_address(first1)};
823 wide y_pred {std::to_address(first2)};
824 wide weight {std::to_address(first3)};
825 we(eve::abs(y_true - y_pred) / eve::max(eps, eve::abs(y_true)), weight);
826 detail::advance(s, first1, first2, first3);
827 }
828
829 // use scalar accumulators for the remaining values
830 auto se = univariate_accumulator<T>::load_state(we.stats());
831 for (; first1 < last1; ++first1, ++first2, ++first3) {
832 se(eve::abs(*first1 - *first2) / eve::max(eps, eve::abs(*first1)), *first3);
833 }
834 return univariate_statistics(se).mean;
835}

924template<std::floating_point T, std::input_iterator I, std::input_iterator J>
925inline auto gaussian_neg_likelihood_loss(I first1, std::sentinel_for<I> auto last1, J first2, T sigma) noexcept
926 -> double
927{
928 using wide = eve::wide<T>;
929 auto constexpr s {wide::size()};

934 for (auto i = 0; i < m; i += s) {
935 wide y_true {std::to_address(first1)};
936 wide y_pred {std::to_address(first2)};
937 we(eve::sqr(y_true - y_pred));
938 detail::advance(s, first1, first2);
939 }
940
941 // use scalar accumulators for the remaining values
942 auto se = univariate_accumulator<T>::load_state(we.stats());
943 for (; first1 < last1; ++first1, ++first2) {
944 se(eve::sqr(*first1 - *first2));
945 }

949 + static_cast<double>(n) * std::log(static_cast<double>(sigma))
950 + ssr / (2.0 * static_cast<double>(sigma) * static_cast<double>(sigma));
951}
